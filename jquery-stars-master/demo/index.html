jQuery stars

A rating star plugin using icon fonts

View on github

More stars

<div class="more-stars"></div>
$(".more-stars").stars({ stars:20 });

Font size

<div class="font-size"></div>
.font-size i { font-size:9px; }

$(".font-size").stars();

Init with values

<div class="value-set"></div>
$(".value-set").stars({ value:4 });

Color change

<div class="green-color"></div>
$(".green-color").stars({ color:'#73AD21' });

Icon change

<div class="icon-change"></div>
$(".icon-change").stars({
        emptyIcon: 'fa-thumbs-o-up',
        filledIcon: 'fa-thumbs-up'
});

With text

<div class="text"></div>
$(".text").stars({ 
        text: ["1 star", "2 star", "3 star", "4 star", "5 star"]
});

Callback

<div class="click-callback"></div>
$(".click-callback").stars({ 
        click: function(i) {
            alert("Star " + i + " selected.");
        }
});
